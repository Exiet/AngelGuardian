## 二、步骤一：BSP 房间生成

### 2.1 算法伪代码

```python
class Room:
    def __init__(self, x, y, w, h):
        self.x = x          # 左上角x
        self.y = y          # 左上角y
        self.w = w          # 宽度
        self.h = h          # 高度
        self.center = (x + w//2, y + h//2)
        self.connected = False

def generate_rooms(map_w, map_h, min_room_size=400, max_depth=5):
    root = Node(0, 0, map_w, map_h)
    rooms = []
    
    # BSP递归分割
    def split(node, depth):
        if depth >= max_depth:
            # 叶节点 = 房间
            if node.w >= min_room_size and node.h >= min_room_size:
                # 在节点内随机生成房间（比节点小，留墙厚）
                margin = 40  # 墙厚
                rw = random.randint(min_room_size, node.w - margin*2)
                rh = random.randint(min_room_size, node.h - margin*2)
                rx = node.x + random.randint(margin, node.w - rw - margin)
                ry = node.y + random.randint(margin, node.h - rh - margin)
                rooms.append(Room(rx, ry, rw, rh))
            return
        # 随机选择横切/竖切
        if node.w > node.h * 1.5:
            orientation = "vertical"
        elif node.h > node.w * 1.5:
            orientation = "horizontal"
        else:
            orientation = random.choice(["horizontal", "vertical"])
        
        # 分割位置（偏移±15%防均匀）
        if orientation == "horizontal":
            split_at = node.y + random.randint(
                int(node.h * 0.35), int(node.h * 0.65)
            )
            node.left = Node(node.x, node.y, node.w, split_at - node.y)
            node.right = Node(node.x, split_at, node.w, node.h - (split_at - node.y))
        else:
            split_at = node.x + random.randint(
                int(node.w * 0.35), int(node.w * 0.65)
            )
            node.left = Node(node.x, node.y, split_at - node.x, node.h)
            node.right = Node(split_at, node.y, node.w - (split_at - node.x), node.h)
        
        split(node.left, depth + 1)
        split(node.right, depth + 1)
    
    split(root, 0)
    
    # 过滤：删除面积过小的房间（但保证 ≥ 8 个）
    min_area = min_room_size * min_room_size * 0.6
    rooms = [r for r in rooms if r.w * r.h >= min_area]
    if len(rooms) < 8:
        # 放宽条件，重新生成（或降低min_room_size）
        pass  # 实际实现中递归重试
    
    return rooms
```

## 三、步骤二：走廊连接

### 3.1 Delaunay 三角剖分 + MST

```python
def connect_rooms(rooms):
    # 1. 对所有房间中心做 Delaunay 三角剖分
    points = [r.center for r in rooms]
    triangles = delaunay_triangulation(points)
    
    # 2. 提取所有边，按距离排序
    edges = []
    for tri in triangles:
        for i in range(3):
            for j in range(i+1, 3):
                p1, p2 = points[i], points[j]
                d = distance(p1, p2)
                edges.append((d, i, j))
    edges.sort()  # 按距离升序
    
    # 3. Kruskal 算法求 MST（最小生成树）
    parent = list(range(len(rooms)))
    def find(x):
        return x if parent[x] == x else find(parent[x])
    
    mst_edges = []
    for d, i, j in edges:
        pi, pj = find(i), find(j)
        if pi != pj:
            parent[pi] = pj
            mst_edges.append((i, j))
            rooms[i].connected = True
            rooms[j].connected = True
    
    # 4. 随机回填 30~50% 的非MST边（形成环路）
    non_mst = [(d,i,j) for (d,i,j) in edges if (i,j) not in mst_edges and (j,i) not in mst_edges]
    random.shuffle(non_mst)
    extra_count = int(len(non_mst) * random.uniform(0.3, 0.5))
    extra_edges = non_mst[:extra_count]
    
    return mst_edges, extra_edges
```

### 3.2 走廊路径生成（L形）

def generate_corridor(room_a, room_b, corridor_width=150):
    # L形走廊：先水平再垂直（或先垂直再水平）
    cx, cy = room_a.center
    dx, dy = room_b.center
    
    if random.choice([True, False]):
        # 先水平后垂直
        waypoint = (dx, cy)
    else:
        # 先垂直后水平
        waypoint = (cx, dy)
    
    # 将走廊区域标记为"可通行"（清除障碍物）
    clear_corridor(cx, cy, waypoint[0], waypoint[1], corridor_width)
    clear_corridor(waypoint[0], waypoint[1], dx, dy, corridor_width)
    
    # 走廊与房间连接处 → 放置门（30%概率）
    if random.random() < 0.3:
        place_door_at(cx, cy, corridor_width)
    if random.random() < 0.3:
        place_door_at(dx, dy, corridor_width)

## 四、步骤三：质量验证与修复

```python
def validate_and_fix(rooms, edges):
    issues = []
    
    # 1. 连通性检查（BFS）
    visited = set()
    queue = [0]  # 从房间0开始
    while queue:
        idx = queue.pop(0)
        if idx in visited:
            continue
        visited.add(idx)
        for (i,j) in edges:
            if i == idx and j not in visited:
                queue.append(j)
            elif j == idx and i not in visited:
                queue.append(i)
    
    if len(visited) < len(rooms):
        issues.append("不连通：有房间无法到达")
        # 修复：在未连通房间和最近已连通房间之间加边
        for i in range(len(rooms)):
            if i not in visited:
                nearest = min(visited, key=lambda j: distance(rooms[i].center, rooms[j].center))
                edges.append((i, nearest))
    
    # 2. 环形走廊检查（拓扑检测）
    # 使用 DFS 检测回边
    has_loop = False
    # ...（图论：如果边数 > 节点数-1，则一定有环）
    if len(edges) <= len(rooms) - 1:
        issues.append("无环路：地图是树结构")
        # 修复：随机加一条非MST边
        all_pairs = [(i,j) for i in range(len(rooms)) for j in range(i+1, len(rooms))]
        random.shuffle(all_pairs)
        for (i,j) in all_pairs:
            if (i,j) not in edges and (j,i) not in edges:
                edges.append((i,j))
                break
    
    # 3. 口袋房间检查（只有1个连接的房间）
    degrees = [0] * len(rooms)
    for (i,j) in edges:
        degrees[i] += 1
        degrees[j] += 1
    pocket_rooms = [i for i, d in enumerate(degrees) if d == 1]
    if len(pocket_rooms) < 1:
        issues.append("无口袋房间")
        # 修复：随机选一个房间，删除其最远连接（使其变成口袋）
        # （实际操作：选一个度≥2的房间，移除一条边）
    
    # 4. 门数量检查
    door_count = count_doors()
    door_rate = door_count / len(edges)
    if door_rate < 0.3:
        issues.append(f"门太少：{door_rate*100:.0f}% < 30%")
        # 修复：在未设门的走廊随机补门
    
    return issues, edges
```

## 六、性能优化建议

| 优化项 | 建议 | 原因 |
|--------|------|------|
| BSP生成 | 只在每局开始时运行一次，缓存结果 | 生成耗时约 10~50ms，不值得每帧计算 |
| 走廊碰撞 | 使用 TileMap 而非精细碰撞 | 俯视角游戏，TileMap 足够 |
| 地形活化检测 | 每 1 秒检测一次（而非每帧） | 阶段性变化，不需要实时 |
| 墙体渲染 | 使用 SpriteBatch 合批绘制 | 墙体数量多（可达数百个Sprite） |
| 门的状态同步 | 门状态变化时才同步（联机模式） | 门开关频率低 |
